# 🎮 MEGA PROMPT — GameReviews Platform (Para Replit)

---

## 🎯 OBJETIVO DEL PROYECTO

Construye una plataforma web completa llamada **GameReviews** — una comunidad donde los usuarios pueden buscar videojuegos, escribir reseñas, votar, construir su lista personal de juegos y ver rankings comunitarios. Todo funciona sin backend real: los datos se guardan en `localStorage` y la IA está integrada mediante la **API de Anthropic** directamente desde el frontend.

## 🤖 INTEGRACIÓN CON ANTHROPIC API

### Configuración:
```javascript
// En utils.js
const ANTHROPIC_API_URL = "https://api.anthropic.com/v1/messages";
const ANTHROPIC_MODEL = "claude-sonnet-4-20250514";

async function llamarIA(systemPrompt, userPrompt, onChunk) {
  const response = await fetch(ANTHROPIC_API_URL, {
    method: "POST",
    headers: {
      "Content-Type": "application/json",
      // La API key se maneja automáticamente en Replit con secrets
    },
    body: JSON.stringify({
      model: ANTHROPIC_MODEL,
      max_tokens: 1000,
      stream: true,
      system: systemPrompt,
      messages: [{ role: "user", content: userPrompt }]
    })
  });

  const reader = response.body.getReader();
  const decoder = new TextDecoder();
  
  while (true) {
    const { done, value } = await reader.read();
    if (done) break;
    
    const chunk = decoder.decode(value);
    const lines = chunk.split('\n').filter(l => l.startsWith('data: '));
    
    for (const line of lines) {
      try {
        const data = JSON.parse(line.slice(6));
        if (data.type === 'content_block_delta') {
          onChunk(data.delta.text);
        }
      } catch (e) {}
    }
  }
}
```

### Configurar la API key en Replit:
1. En Replit, ir a "Secrets" (candado en el sidebar)
2. Crear secret: `ANTHROPIC_API_KEY` = tu key
3. Acceder con: `process.env.ANTHROPIC_API_KEY` (si usas un servidor) o configurar un proxy simple

> **IMPORTANTE**: Si es pure frontend sin servidor, crear un archivo `server.js` con Express como proxy que reciba las peticiones del frontend y las reenvíe a Anthropic con la API key desde las variables de entorno de Replit.

### Proxy server simple (`server.js`):
```javascript
const express = require('express');
const fetch = require('node-fetch');
const app = express();

app.use(express.static('.'));
app.use(express.json());

app.post('/api/ia', async (req, res) => {
  const response = await fetch('https://api.anthropic.com/v1/messages', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'x-api-key': process.env.ANTHROPIC_API_KEY,
      'anthropic-version': '2023-06-01'
    },
    body: JSON.stringify(req.body)
  });
  
  res.setHeader('Content-Type', 'text/event-stream');
  response.body.pipe(res);
});

app.listen(3000);
```

### Sistema de reseñas:
```javascript
// Guardar reseña
function publicarReseña(juegoId, rating, texto, recomendado) {
  const usuario = getUsuario();
  const reseña = {
    id: `rev_${Date.now()}`,
    juegoId,
    autor: usuario.nombre,
    rating,
    texto,
    recomendado,
    fecha: new Date().toISOString(),
    utilidad: 0,
    esPropia: true
  };
  
  const reseñas = getReseñas();
  reseñas.push(reseña);
  localStorage.setItem('gr_reseñas', JSON.stringify(reseñas));
  return reseña;
}

// Calcular rating promedio de un juego
function calcularRatingComunidad(juegoId) {
  const reseñas = getReseñas().filter(r => r.juegoId === juegoId);
  if (!reseñas.length) return null;
  const suma = reseñas.reduce((s, r) => s + r.rating, 0);
  return (suma / reseñas.length).toFixed(1);
}
```

### Buscador con debounce:
```javascript
let searchTimeout;
document.getElementById('buscador').addEventListener('input', (e) => {
  clearTimeout(searchTimeout);
  searchTimeout = setTimeout(() => buscar(e.target.value), 300);
});

function buscar(query) {
  const q = query.toLowerCase().trim();
  const resultados = todosLosJuegos.filter(j => 
    j.nombre.toLowerCase().includes(q) ||
    j.categoria.toLowerCase().includes(q) ||
    j.desarrollador.toLowerCase().includes(q) ||
    j.tags?.some(t => t.includes(q))
  );
  renderJuegos(resultados);
}
```

### Toast notifications:
```javascript
function mostrarToast(mensaje, tipo = 'success') {
  // tipo: 'success' | 'error' | 'info' | 'warning'
  const colores = {
    success: '#10b981',
    error: '#ef4444', 
    info: '#3b82f6',
    warning: '#f59e0b'
  };
  
  const toast = document.createElement('div');
  toast.className = 'toast';
  toast.style.borderLeftColor = colores[tipo];
  toast.innerHTML = `<span>${mensaje}</span>`;
  document.body.appendChild(toast);
  
  setTimeout(() => toast.classList.add('show'), 10);
  setTimeout(() => {
    toast.classList.remove('show');
    setTimeout(() => toast.remove(), 300);
  }, 3000);
}
```

## ✨ DETALLES EXTRAS QUE ELEVAN LA CALIDAD

1. **Skeleton loading**: Mostrar placeholders animados mientras cargan los juegos
2. **Infinite scroll** o paginación (mostrar 12 juegos, "Cargar más" al final)
3. **Modo sin JS**: Mensaje explicativo
4. **Tooltips** en botones con `title` o custom CSS
5. **Confirmación al eliminar**: Modal pequeño antes de eliminar reseña
6. **Animación de rating**: Al hacer hover sobre las estrellas, se llenan de color
7. **Scroll to top** botón flotante al hacer scroll hacia abajo
8. **Indicador de "Nuevo"**: Badge en juegos lanzados hace menos de 30 días
9. **Compartir**: Botón de share nativo (`navigator.share`) para compartir la URL del juego
10. **Accesibilidad**: `aria-label` en botones icon-only, contraste adecuado, focus visible
