Ve a https://isthereanydeal.com/api/ y usa esta key de ejemplo o registra una nueva.

### Nuevo endpoint en server.js (o el backend existente):

```javascript
// Proxy para IsThereAnyDeal - evitar CORS
app.get('/api/deals/:slug', async (req, res) => {
  try {
    const { slug } = req.params;
    const apiKey = process.env.ITAD_API_KEY; // agregar a Secrets de Replit

    // Buscar el juego por nombre
    const searchRes = await fetch(
      `https://api.isthereanydeal.com/games/search/v1?key=${apiKey}&title=${encodeURIComponent(slug)}&results=1`
    );
    const searchData = await searchRes.json();
    
    if (!searchData?.length) return res.json({ deals: [], found: false });
    
    const gameId = searchData[0].id;

    // Obtener precios actuales en todas las tiendas
    const pricesRes = await fetch(
      `https://api.isthereanydeal.com/games/prices/v2?key=${apiKey}`,
      {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify([gameId])
      }
    );
    const pricesData = await pricesRes.json();
    const prices = pricesData?.[0]?.deals || [];

    // Obtener precio histórico mínimo
    const histRes = await fetch(
      `https://api.isthereanydeal.com/games/historylow/v1?key=${apiKey}`,
      {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify([gameId])
      }
    );
    const histData = await histRes.json();
    const historicalLow = histData?.[0]?.low || null;

    res.json({
      found: true,
      gameId,
      deals: prices.map(d => ({
        tienda: d.shop.name,
        precio: d.price.amount,
        precioOriginal: d.regular.amount,
        descuento: Math.round(((d.regular.amount - d.price.amount) / d.regular.amount) * 100),
        url: d.url,
        enOferta: d.price.amount < d.regular.amount
      })),
      precioHistoricoMinimo: historicalLow?.amount || null,
      moneda: 'USD'
    });
  } catch (e) {
    res.json({ deals: [], found: false, error: e.message });
  }
});

// Endpoint para los juegos más baratos del momento (para la home)
app.get('/api/deals/top/ofertas', async (req, res) => {
  try {
    const apiKey = process.env.ITAD_API_KEY;
    const response = await fetch(
      `https://api.isthereanydeal.com/deals/v2?key=${apiKey}&limit=10&sort=price:asc`
    );
    const data = await response.json();
    res.json(data);
  } catch (e) {
    res.json({ list: [] });
  }
});
```

```javascript
// Al cargar la página del juego, llamar automáticamente
async function cargarPrecios(nombreJuego) {
  const seccion = document.getElementById('precios-section');
  seccion.innerHTML = '<div class="skeleton">Buscando precios...</div>';
  
  try {
    const res = await fetch(`/api/deals/${encodeURIComponent(nombreJuego)}`);
    const data = await res.json();
    
    if (!data.found || !data.deals.length) {
      seccion.innerHTML = '<p class="muted">No se encontraron precios disponibles</p>';
      return;
    }
    
    // Ordenar: primero los que tienen descuento
    const ordenados = data.deals.sort((a, b) => b.descuento - a.descuento);
    
    const iconos = {
      'Steam': '🎮', 'PlayStation': '🎮', 'Xbox': '🟢',
      'GOG': '🔵', 'Epic Games': '⚫', 'Nintendo': '🔴',
      'Humble Store': '🟠', 'Fanatical': '🟣'
    };
    
    seccion.innerHTML = `
      <div class="precios-grid">
        ${ordenados.map(d => `
          <div class="precio-row ${d.enOferta ? 'en-oferta' : ''}">
            <span class="tienda-nombre">${iconos[d.tienda] || '🛒'} ${d.tienda}</span>
            <div class="precio-info">
              ${d.enOferta ? `<span class="precio-original">$${d.precioOriginal}</span>` : ''}
              <span class="precio-actual ${d.enOferta ? 'oferta' : ''}">$${d.precio}</span>
              ${d.enOferta ? `<span class="badge-descuento">-${d.descuento}%</span>` : ''}
            </div>
            <a href="${d.url}" target="_blank" class="btn-ver-oferta">
              ${d.enOferta ? '🔥 Ver oferta' : 'Ver precio'}
            </a>
          </div>
        `).join('')}
        ${data.precioHistoricoMinimo ? `
          <div class="precio-historico">
            📉 Precio histórico mínimo: <strong>$${data.precioHistoricoMinimo}</strong>
          </div>
        ` : ''}
      </div>
    `;
  } catch (e) {
    seccion.innerHTML = '<p class="muted">Error al cargar precios</p>';
  }
}
```

### Lógica del plan (localStorage por ahora):
```javascript
// En utils.js
function getPlan() {
  return localStorage.getItem('gr_plan') || 'gratis'; // 'gratis' | 'pro' | 'elite'
}

function setPlan(plan) {
  localStorage.setItem('gr_plan', plan);
  localStorage.setItem('gr_plan_fecha', new Date().toISOString());
}

// Límites por plan
const LIMITES = {
  gratis: {
    reseñasPorMes: 5,
    juegosEnLista: 10,
    analisisIA: 3,
    alertasOfertas: false,
    sinAnuncios: false,
    badge: null
  },
  pro: {
    reseñasPorMes: Infinity,
    juegosEnLista: Infinity,
    analisisIA: Infinity,
    alertasOfertas: true,
    sinAnuncios: true,
    badge: '⭐ PRO'
  },
  elite: {
    reseñasPorMes: Infinity,
    juegosEnLista: Infinity,
    analisisIA: Infinity,
    alertasOfertas: true,
    sinAnuncios: true,
    badge: '👑 ELITE',
    apiAcceso: true,
    reporteMensual: true
  }
};

function puedeEscribirReseña() {
  const plan = getPlan();
  if (LIMITES[plan].reseñasPorMes === Infinity) return true;
  
  const reseñas = getReseñasDelMes(); // filtra por mes actual
  return reseñas.length < LIMITES[plan].reseñasPorMes;
}

function puedeUsarIA() {
  const plan = getPlan();
  if (LIMITES[plan].analisisIA === Infinity) return true;
  
  const usos = parseInt(localStorage.getItem('gr_ia_usos_hoy') || '0');
  return usos < LIMITES[plan].analisisIA;
}
```

## 6. 🔔 ALERTAS DE OFERTAS (solo plan PRO/ELITE)

En la página de cada juego, si el usuario tiene plan PRO o ELITE:

```javascript
function activarAlertaOferta(juegoId, nombreJuego) {
  if (!['pro', 'elite'].includes(getPlan())) {
    mostrarModalUpgrade(
      '🔔 Alertas de ofertas',
      'Recibe una notificación cuando este juego esté en oferta. Función exclusiva PRO.',
      'pro'
    );
    return;
  }
  
  const alertas = JSON.parse(localStorage.getItem('gr_alertas') || '[]');
  const yaExiste = alertas.find(a => a.juegoId === juegoId);
  
  if (yaExiste) {
    mostrarToast('🔕 Alerta desactivada', 'info');
    localStorage.setItem('gr_alertas', JSON.stringify(alertas.filter(a => a.juegoId !== juegoId)));
  } else {
    alertas.push({ juegoId, nombreJuego, fechaActivada: new Date().toISOString() });
    localStorage.setItem('gr_alertas', JSON.stringify(alertas));
    mostrarToast('🔔 Te avisaremos cuando baje de precio', 'success');
  }
}
```

```javascript
app.get('/api/deals/history/:gameId', async (req, res) => {
```
