## 1. 🗄️ BASE DE DATOS REAL (SQLite + mejor-sqlite3)

El proyecto actualmente usa localStorage, lo que significa que cada usuario solo ve SUS propias reseñas.
Para que sea una comunidad real donde TODOS vean las reseñas de TODOS, necesitamos una base de datos real en el servidor.

### Instalar:
```bash
npm install better-sqlite3
npm install @types/better-sqlite3
```

### Crear archivo `server/database.js`:

```javascript
const Database = require('better-sqlite3');
const path = require('path');

const db = new Database(path.join(__dirname, 'gamereviews.db'));

// Crear tablas si no existen
db.exec(`
  CREATE TABLE IF NOT EXISTS resenas (
    id TEXT PRIMARY KEY,
    juego_id INTEGER NOT NULL,
    juego_nombre TEXT NOT NULL,
    autor TEXT NOT NULL,
    rating INTEGER NOT NULL CHECK(rating >= 1 AND rating <= 5),
    texto TEXT NOT NULL,
    recomendado INTEGER NOT NULL DEFAULT 1,
    fecha TEXT NOT NULL,
    utilidad INTEGER NOT NULL DEFAULT 0,
    plataforma TEXT DEFAULT 'PC'
  );

  CREATE TABLE IF NOT EXISTS votos_utilidad (
    resena_id TEXT NOT NULL,
    usuario_hash TEXT NOT NULL,
    PRIMARY KEY (resena_id, usuario_hash)
  );

  CREATE TABLE IF NOT EXISTS visitas_juego (
    juego_id INTEGER PRIMARY KEY,
    contador INTEGER NOT NULL DEFAULT 0,
    ultima_visita TEXT
  );

  CREATE TABLE IF NOT EXISTS alertas_precio (
    juego_id INTEGER NOT NULL,
    juego_nombre TEXT NOT NULL,
    usuario_hash TEXT NOT NULL,
    fecha_activada TEXT NOT NULL,
    PRIMARY KEY (juego_id, usuario_hash)
  );
`);

module.exports = db;
```

### Endpoints nuevos en `server.js`:

```javascript
const db = require('./server/database');

// ===== RESEÑAS =====

// Obtener todas las reseñas de un juego
app.get('/api/resenas/:juegoId', (req, res) => {
  const { juegoId } = req.params;
  const resenas = db.prepare(
    'SELECT * FROM resenas WHERE juego_id = ? ORDER BY fecha DESC'
  ).all(juegoId);
  res.json(resenas);
});

// Obtener TODAS las reseñas (para ranking y estadísticas)
app.get('/api/resenas', (req, res) => {
  const resenas = db.prepare(
    'SELECT * FROM resenas ORDER BY fecha DESC LIMIT 100'
  ).all();
  res.json(resenas);
});

// Crear nueva reseña
app.post('/api/resenas', (req, res) => {
  const { id, juego_id, juego_nombre, autor, rating, texto, recomendado, fecha, plataforma } = req.body;

  if (!texto || texto.length < 10) {
    return res.status(400).json({ error: 'La reseña debe tener al menos 10 caracteres' });
  }
  if (!rating || rating < 1 || rating > 5) {
    return res.status(400).json({ error: 'El rating debe ser entre 1 y 5' });
  }

  try {
    db.prepare(`
      INSERT INTO resenas (id, juego_id, juego_nombre, autor, rating, texto, recomendado, fecha, plataforma)
      VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)
    `).run(id, juego_id, juego_nombre, autor, rating, texto, recomendado ? 1 : 0, fecha, plataforma || 'PC');
    res.json({ success: true, id });
  } catch (e) {
    res.status(500).json({ error: e.message });
  }
});

// Eliminar reseña (solo el autor puede)
app.delete('/api/resenas/:id', (req, res) => {
  const { id } = req.params;
  const { autor } = req.body;
  const resena = db.prepare('SELECT * FROM resenas WHERE id = ?').get(id);
  if (!resena) return res.status(404).json({ error: 'Reseña no encontrada' });
  if (resena.autor !== autor) return res.status(403).json({ error: 'No autorizado' });
  db.prepare('DELETE FROM resenas WHERE id = ?').run(id);
  res.json({ success: true });
});

// Votar utilidad de reseña
app.post('/api/resenas/:id/utilidad', (req, res) => {
  const { id } = req.params;
  const { usuarioHash } = req.body;
  try {
    db.prepare('INSERT INTO votos_utilidad (resena_id, usuario_hash) VALUES (?, ?)').run(id, usuarioHash);
    db.prepare('UPDATE resenas SET utilidad = utilidad + 1 WHERE id = ?').run(id);
    res.json({ success: true });
  } catch (e) {
    res.status(400).json({ error: 'Ya votaste esta reseña' });
  }
});

// ===== ESTADÍSTICAS =====

// Rating promedio de un juego (calculado de la BD)
app.get('/api/stats/juego/:juegoId', (req, res) => {
  const { juegoId } = req.params;
  const stats = db.prepare(`
    SELECT 
      COUNT(*) as total_resenas,
      AVG(rating) as rating_promedio,
      SUM(CASE WHEN recomendado = 1 THEN 1 ELSE 0 END) as total_recomendados
    FROM resenas WHERE juego_id = ?
  `).get(juegoId);
  res.json(stats);
});

// Top juegos más reseñados (para ranking)
app.get('/api/stats/ranking', (req, res) => {
  const ranking = db.prepare(`
    SELECT 
      juego_id,
      juego_nombre,
      COUNT(*) as total_resenas,
      AVG(rating) as rating_promedio,
      SUM(utilidad) as total_utilidad
    FROM resenas
    GROUP BY juego_id
    ORDER BY total_resenas DESC, rating_promedio DESC
    LIMIT 20
  `).all();
  res.json(ranking);
});

// ===== VISITAS =====
app.post('/api/visitas/:juegoId', (req, res) => {
  const { juegoId } = req.params;
  db.prepare(`
    INSERT INTO visitas_juego (juego_id, contador, ultima_visita)
    VALUES (?, 1, datetime('now'))
    ON CONFLICT(juego_id) DO UPDATE SET
      contador = contador + 1,
      ultima_visita = datetime('now')
  `).run(juegoId);
  res.json({ success: true });
});

app.get('/api/visitas', (req, res) => {
  const visitas = db.prepare('SELECT * FROM visitas_juego ORDER BY contador DESC').all();
  res.json(visitas);
});
```

### Actualizar el frontend para usar la BD:

En lugar de guardar reseñas en localStorage, el frontend debe:
1. Al cargar la página de un juego → `GET /api/resenas/:juegoId`
2. Al publicar una reseña → `POST /api/resenas`
3. Al eliminar → `DELETE /api/resenas/:id`
4. Al votar útil → `POST /api/resenas/:id/utilidad`
5. Las reseñas del usuario actual se identifican comparando el nombre guardado en localStorage

**IMPORTANTE:** Mantener localStorage solo para datos personales del usuario (nombre, Mi Lista, plan, etc.) La BD es solo para reseñas compartidas.

## 3. 🔗 ENLACES DE COMPRA EN PÁGINA DE JUEGO

En la página de detalle de cada juego, agrega una sección **"¿Dónde comprarlo?"** con botones que redirigen a las tiendas oficiales:

```
┌─────────────────────────────────────────────────────┐
│  🛒 ¿Dónde comprarlo?                               │
├─────────────────────────────────────────────────────┤
│  🎮 Steam          [Ver en Steam →]                 │
│  🔵 PlayStation    [Ver en PS Store →]              │
│  🟢 Xbox           [Ver en Xbox Store →]            │
│  🔴 Nintendo       [Ver en Nintendo Store →]        │
│  ⚫ Epic Games     [Ver en Epic →]                  │
│                                                     │
│  ⚠️ GameReviews no vende juegos. Te redirigimos    │
│     a las tiendas oficiales.                        │
└─────────────────────────────────────────────────────┘
```

CSS para los botones:
```css
.tienda-btn {
  display: flex;
  align-items: center;
  gap: 10px;
  padding: 10px 20px;
  border-radius: 8px;
  text-decoration: none;
  font-weight: 600;
  transition: all 0.2s;
  border: 1px solid rgba(255,255,255,0.1);
}
.tienda-btn:hover { transform: translateY(-2px); opacity: 0.9; }
.tienda-btn.steam { background: #1b2838; color: #c7d5e0; }
.tienda-btn.playstation { background: #003791; color: white; }
.tienda-btn.xbox { background: #107C10; color: white; }
.tienda-btn.nintendo { background: #E4000F; color: white; }
.tienda-btn.epic { background: #2a2a2a; color: white; }
```

Lógica:
```javascript
function renderEnlacesCompra(juego) {
  const tiendas = juego.enlacesCompra || {};
  const iconos = {
    steam: '🎮', playstation: '🔵', xbox: '🟢',
    nintendo: '🔴', epic: '⚫', microsoft: '🟢'
  };
  const nombres = {
    steam: 'Steam', playstation: 'PlayStation Store',
    xbox: 'Xbox Store', nintendo: 'Nintendo Store',
    epic: 'Epic Games', microsoft: 'Microsoft Store'
  };

  const enlaces = Object.entries(tiendas)
    .filter(([_, url]) => url !== null)
    .map(([tienda, url]) => `
      <a href="${url}" target="_blank" rel="noopener noreferrer"
         class="tienda-btn ${tienda}">
        ${iconos[tienda] || '🛒'} ${nombres[tienda] || tienda}
        <span style="margin-left:auto">→</span>
      </a>
    `).join('');

  return enlaces.length > 0 ? `
    <div class="enlaces-compra">
      <h3 class="section-title">🛒 ¿Dónde comprarlo?</h3>
      <div class="tiendas-grid">${enlaces}</div>
      <p class="aviso-legal">
        ⚠️ GameReviews no vende juegos. Los enlaces te redirigen a las tiendas oficiales.
      </p>
    </div>
  ` : '';
}
```
